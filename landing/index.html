Quick Start

# Convert HTML to PDF
curl -X POST https://api.pdfping.dev/api/v1/convert \
-H "Authorization: Bearer YOUR_API_KEY" \
-H "Content-Type: application/json" \
-d '{ "html": "<h1>Hello World</h1><p>PDF generated!</p>" }' \
-o output.pdf

# Or convert a URL
curl -X POST https://api.pdfping.dev/api/v1/convert \
-H "Authorization: Bearer YOUR_API_KEY" \
-H "Content-Type: application/json" \
-d '{ "url": "https://example.com" }' \
-o page.pdf
