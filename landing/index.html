API
# Convert HTML to PDF
curl -X POST https://pdfapi.uhadev.com/api/v1/convert \
-H "Authorization: Bearer YOUR_API_KEY" \
-H "Content-Type: application/json" \
-d '{ "html": "<h1>Hello World</h1>" }' \
-o output.pdf

# Or convert a URL
curl -X POST https://pdfapi.uhadev.com/api/v1/convert \
-H "Authorization: Bearer YOUR_API_KEY" \
-H "Content-Type: application/json" \
-d '{ "url": "https://example.com" }' \
-o page.pdf
